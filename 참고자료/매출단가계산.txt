-- 매출단가가 0이면
-- 매출단가 = 매입단가 * 1.2(20%마진계산)
-- 매출단가 = 매출단가 * 1.1 : 부가세 포함
-- 원단위가 0이 아닌 값이 나온다
-- INT(매출단가 / 10) : 소수점이하를 자른 값
-- 매출단가 = 매출단가 * 10

-- 매출단가 0 이면 매입단가 * 1.2
-- 매출단가 = IF(C2 = 0, B2*1.2,C2)
-- 매출단가 = 매출단가 * 1.1
-- 매출단가 = INT(매출단가 / 10)
-- 매출단가 = 매출단가 * 10

-- 매입단가 계산식
	=IF(B2=0,INT((C2*0.8)/1.1),B2)

-- 매출단가 계산식
	=IF(C2=0, INT(((B2*1.2)*1.1)/10)*10,C2)
